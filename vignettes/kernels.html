real_kernels <- c("normal", "laplace", "cauchy")

for (ker in real_kernels) {
  for (backend in backends) {
    bundle <- build_nimble_bundle(

      backend = backend,
      kernel = ker,
      GPD = FALSE,
      components = nc_real200_k2$meta$K_true,
      mcmc = mcmc
    )
    summary(bundle)
    fit <- run_mcmc_bundle_manual(bundle, show_progress = FALSE)
    summary(fit)
  }
}
